from flask import Flask, render_template, request, redirect, session, url_for  # Flask-Werkzeuge für Webseiten, Formulare, Login und Weiterleitungen
import sqlite3  # SQLite wird für die Datenbank verwendet
from database import create_database  # importiert unsere Funktion aus database.py


app = Flask(__name__)  # erstellt die Flask-Webanwendung
app.secret_key = "secret_key_for_login"  # wird benötigt, damit Flask eingeloggte Benutzer speichern kann

create_database()  # stellt sicher, dass die Datenbank und Tabellen existieren


def get_db_connection():
    connection = sqlite3.connect("hospital.db")  # verbindet sich mit der Datenbankdatei
    connection.row_factory = sqlite3.Row  # erlaubt Zugriff auf Spaltennamen, z. B. user["role"]
    return connection  # gibt die Datenbankverbindung zurück

def get_role_label(role):
    # Übersetzt technische Rollennamen in verständliche deutsche Namen
    role_labels = {
        "doctor": "Arzt / Ärztin",
        "nurse": "Pflegekraft",
        "lab": "Laborpersonal",
        "admin": "Administrator"
    }

    return role_labels.get(role, role)


def safe_count(query):
    # Führt eine Zählabfrage sicher aus
    # Falls eine Tabelle noch nicht existiert, wird 0 zurückgegeben
    try:
        connection = get_db_connection()
        result = connection.execute(query).fetchone()[0]
        connection.close()
        return result
    except sqlite3.OperationalError:
        return 0


def has_permission(role, data_type):
    # Hier wird festgelegt, welche Rolle welche Patientendaten sehen darf
    permissions = {
        "doctor": ["basic", "diagnosis", "medication", "lab_results", "vital_values"],
        "nurse": ["basic", "medication", "vital_values"],
        "lab": ["basic", "lab_results"],
        "admin": ["basic", "diagnosis", "medication", "lab_results", "vital_values", "logs"]
    }

    return data_type in permissions.get(role, [])  # gibt True zurück, wenn die Rolle Zugriff hat


def log_access(username, role, patient_name, accessed_data, access_reason):
    connection = get_db_connection()  # öffnet Datenbankverbindung

    # speichert den Zugriff inklusive Zugriffsgrund
    connection.execute(""" 
        INSERT INTO access_logs (username, role, patient_name, accessed_data, access_reason) 
        VALUES (?, ?, ?, ?, ?) 
    """, (username, role, patient_name, accessed_data, access_reason))

    connection.commit()  # speichert die Änderung
    connection.close()  # schließt Datenbankverbindung

@app.route("/login", methods=["GET", "POST"])  # Login-Seite, nimmt GET und POST an
def login():
    error = None  # speichert eine Fehlermeldung, falls Login falsch ist

    if request.method == "POST":  # wird ausgeführt, wenn das Login-Formular abgeschickt wurde
        username = request.form["username"]  # liest den Benutzernamen aus dem Formular
        password = request.form["password"]  # liest das Passwort aus dem Formular

        connection = get_db_connection()  # öffnet Datenbankverbindung

        # sucht einen Benutzer mit passendem Benutzernamen und Passwort
        user = connection.execute(""" 
            SELECT * FROM users 
            WHERE username = ? AND password = ? AND is_active = 1 
        """, (username, password)).fetchone()

        connection.close()  # schließt die Datenbankverbindung

        if user:  # wenn ein passender Benutzer gefunden wurde
            session["username"] = user["username"]  # speichert den Benutzernamen in der Sitzung
            session["role"] = user["role"]  # speichert die Rolle in der Sitzung
            return redirect(url_for("dashboard"))  # leitet zum Dashboard weiter

        error = "Falscher Benutzername oder falsches Passwort."  # Fehlermeldung bei falschem Login

    return render_template("login.html", error=error)  # zeigt die Login-Seite an


@app.route("/dashboard")  # Dashboard-Seite
def dashboard():
    if "username" not in session:  # prüft, ob Benutzer eingeloggt ist
        return redirect(url_for("login"))  # falls nicht, zurück zum Login

    notifications = []  # offene Arzt-Benachrichtigungen
    role = session["role"]  # aktuelle Rolle

    # Statistiken für das Dashboard
    patient_count = safe_count("SELECT COUNT(*) FROM patients")
    user_count = safe_count("SELECT COUNT(*) FROM users")
    open_orders_count = safe_count("SELECT COUNT(*) FROM doctor_orders WHERE status = 'offen'")
    open_medications_count = safe_count("SELECT COUNT(*) FROM medication_plan WHERE status = 'offen'")
    open_notifications_count = safe_count("SELECT COUNT(*) FROM doctor_notifications WHERE status = 'offen'")
    access_logs_count = safe_count("SELECT COUNT(*) FROM access_logs")

    # Ärzte und Admins sehen offene Benachrichtigungen
    if role in ["doctor", "admin"]:
        connection = get_db_connection()

        try:
            notifications = connection.execute(""" 
                SELECT doctor_notifications.*, patients.name AS patient_name 
                FROM doctor_notifications 
                JOIN patients ON doctor_notifications.patient_id = patients.id 
                WHERE doctor_notifications.status = 'offen' 
                ORDER BY doctor_notifications.created_at DESC 
            """).fetchall()
        except sqlite3.OperationalError:
            notifications = []

        connection.close()

    return render_template(
        "dashboard.html",
        username=session["username"],
        role=role,
        role_label=get_role_label(role),
        notifications=notifications,
        patient_count=patient_count,
        user_count=user_count,
        open_orders_count=open_orders_count,
        open_medications_count=open_medications_count,
        open_notifications_count=open_notifications_count,
        access_logs_count=access_logs_count
    )

        "patients.html",
        patients=patient_list,  # gibt Patientenliste an HTML weiter

@app.route("/patient/<int:patient_id>", methods=["GET", "POST"])  # Detailseite mit Zugriffsgrund
def patient_detail(patient_id):
    if "username" not in session:  # prüft, ob Benutzer eingeloggt ist
        return redirect(url_for("login"))  # falls nicht, zurück zum Login

    connection = get_db_connection()  # öffnet Datenbankverbindung

    patient = connection.execute(""" 
        SELECT * FROM patients WHERE id = ? 
    """, (patient_id,)).fetchone()  # lädt Patient anhand der ID

    connection.close()  # schließt Datenbankverbindung

    if patient is None:  # falls Patient nicht existiert
        return "Patient wurde nicht gefunden."

    visible_data = None  # am Anfang werden noch keine Patientendaten angezeigt
    access_reason = None  # am Anfang gibt es noch keinen Zugriffsgrund

    if request.method == "POST":  # wenn der Benutzer den Grund abgeschickt hat
        access_reason = request.form["access_reason"]  # liest den Zugriffsgrund aus dem Formular

        role = session["role"]  # liest Rolle
        username = session["username"]  # liest Benutzername

        visible_data = {}  # hier kommen nur erlaubte Daten rein

        if has_permission(role, "basic"):  # prüft Zugriff auf Stammdaten
            visible_data["Name"] = patient["name"]
            visible_data["Geburtsdatum"] = patient["birthdate"]
            log_access(username, role, patient["name"], "Stammdaten", access_reason)

        if has_permission(role, "diagnosis"):  # prüft Zugriff auf Diagnose
            visible_data["Diagnose"] = patient["diagnosis"]
            log_access(username, role, patient["name"], "Diagnose", access_reason)

        if has_permission(role, "medication"):  # prüft Zugriff auf Medikation
            visible_data["Medikation"] = patient["medication"]
            log_access(username, role, patient["name"], "Medikation", access_reason)

        if has_permission(role, "lab_results"):  # prüft Zugriff auf Laborwerte
            visible_data["Laborwerte"] = patient["lab_results"]
            log_access(username, role, patient["name"], "Laborwerte", access_reason)

        if has_permission(role, "vital_values"):  # prüft Zugriff auf Vitalwerte
            visible_data["Vitalwerte"] = patient["vital_values"]
            log_access(username, role, patient["name"], "Vitalwerte", access_reason)

    connection = get_db_connection()  # öffnet Datenbankverbindung

    nursing_notes = connection.execute(""" 
        SELECT * FROM nursing_notes 
        WHERE patient_id = ? 
        ORDER BY created_at DESC 
    """, (patient_id,)).fetchall()  # lädt Pflegeeinträge des Patienten

    doctor_orders = connection.execute(""" 
        SELECT * FROM doctor_orders 
        WHERE patient_id = ? 
        ORDER BY created_at DESC 
    """, (patient_id,)).fetchall()  # lädt ärztliche Anordnungen des Patienten

    lab_reports = connection.execute(""" 
        SELECT * FROM lab_reports 
        WHERE patient_id = ? 
        ORDER BY created_at DESC 
    """, (patient_id,)).fetchall()  # lädt Laborbefunde des Patienten

    connection.close()  # schließt Datenbankverbindung

    return render_template(
        "patient_detail.html",
        patient=patient,
        visible_data=visible_data,
        access_reason=access_reason,
        nursing_notes=nursing_notes,
        doctor_orders=doctor_orders,
        lab_reports=lab_reports,
        role=session["role"]
    )


@app.route("/logs")  # Seite für Zugriffsprotokolle mit Filterfunktion
def logs():
    if "username" not in session:  # prüft, ob Benutzer eingeloggt ist
        return redirect(url_for("login"))  # falls nicht, zurück zum Login

    if session["role"] != "admin":  # nur Admins dürfen die Logs sehen
        return "Zugriff verweigert. Nur Admins dürfen die Protokolle sehen."

    search_query = request.args.get("search", "")  # liest Suchbegriff aus der URL
    role_filter = request.args.get("role", "")  # liest Rollenfilter aus der URL

    connection = get_db_connection()  # öffnet Datenbankverbindung

    query = """ 
        SELECT * FROM access_logs 
        WHERE 1 = 1 
    """  # Grundabfrage, die später erweitert wird

    parameters = []  # hier speichern wir die Werte für die Platzhalter

    if search_query:  # wenn ein Suchbegriff eingegeben wurde
        query += """ 
            AND ( 
                username LIKE ? 
                OR patient_name LIKE ? 
                OR accessed_data LIKE ? 
                OR access_reason LIKE ? 
            ) 
        """
        parameters.extend([
            f"%{search_query}%",
            f"%{search_query}%",
            f"%{search_query}%",
            f"%{search_query}%"
        ])  # fügt Suchwerte sicher zur Abfrage hinzu

    if role_filter:  # wenn eine Rolle ausgewählt wurde
        query += " AND role = ?"
        parameters.append(role_filter)  # fügt Rollenfilter hinzu

    query += " ORDER BY access_time DESC"  # neueste Einträge zuerst

    log_list = connection.execute(query, parameters).fetchall()  # führt die gefilterte Suche aus
    connection.close()  # schließt Datenbankverbindung

    return render_template(
        "logs.html",
        logs=log_list,  # gibt Log-Einträge an HTML weiter
        search_query=search_query,  # gibt Suchbegriff an HTML weiter
        role_filter=role_filter  # gibt ausgewählte Rolle an HTML weiter
    )

@app.route("/add_user", methods=["GET", "POST"])  # Seite zum Anlegen neuer Benutzer
def add_user():
    if "username" not in session:  # prüft, ob jemand eingeloggt ist
        return redirect(url_for("login"))  # falls nicht, zurück zum Login

    if session["role"] != "admin":  # nur Admins dürfen neue Benutzer anlegen
        return "Zugriff verweigert. Nur Admins dürfen neue Benutzer anlegen."

    message = None  # speichert eine Erfolgsmeldung oder Fehlermeldung

    if request.method == "POST":  # wird ausgeführt, wenn das Formular abgeschickt wurde
        username = request.form["username"]  # liest den neuen Benutzernamen aus
        password = request.form["password"]  # liest das neue Passwort aus
        role = request.form["role"]  # liest die ausgewählte Rolle aus

        connection = get_db_connection()  # öffnet Datenbankverbindung

        try:
            # fügt den neuen Benutzer in die Tabelle users ein
            connection.execute(""" 
                INSERT INTO users (username, password, role)

            """, (username, password, role))

            connection.commit()  # speichert die Änderung
            message = "Benutzer wurde erfolgreich angelegt."  # Erfolgsmeldung

        except sqlite3.IntegrityError:
            # passiert, wenn der Benutzername schon existiert
            message = "Dieser Benutzername existiert bereits."

        connection.close()  # schließt Datenbankverbindung

    return render_template("add_user.html", message=message)  # zeigt die HTML-Seite an

@app.route("/toggle_user/<int:user_id>", methods=["POST"])  # aktiviert oder deaktiviert einen Benutzer
def toggle_user(user_id):

        return redirect(url_for("login"))  # falls nicht, zurück zum Login

    if session["role"] != "admin":  # nur Admins dürfen Benutzer aktivieren/deaktivieren
        return "Zugriff verweigert. Nur Admins dürfen Benutzer aktivieren oder deaktivieren."

    connection = get_db_connection()  # öffnet Datenbankverbindung

    user = connection.execute(""" 
        SELECT * FROM users WHERE id = ? 
    """, (user_id,)).fetchone()  # lädt den ausgewählten Benutzer

    if user is None:  # falls die ID nicht existiert
        connection.close()
        return "Benutzer wurde nicht gefunden."

    if user["username"] == session["username"]:  # Admin soll sich nicht selbst deaktivieren
        connection.close()
        return "Du kannst deinen eigenen Benutzer nicht deaktivieren."

    new_status = 0 if user["is_active"] == 1 else 1  # wechselt zwischen aktiv und deaktiviert

    connection.execute(""" 
        UPDATE users 
        SET is_active = ? 
        WHERE id = ? 
    """, (new_status, user_id))  # speichert den neuen Status

    connection.commit()  # speichert Änderung
    connection.close()  # schließt Datenbankverbindung

    return redirect(url_for("users"))  # zurück zur Benutzerverwaltung

@app.route("/patient_curve/<int:patient_id>", methods=["GET", "POST"])  # Patientenkurve eines Patienten
def patient_curve(patient_id):
    if "username" not in session:
        return redirect(url_for("login"))

    if session["role"] not in ["nurse", "doctor", "admin"]:
        return "Zugriff verweigert. Sie dürfen die Patientenkurve nicht öffnen."

    selected_date = request.args.get("date", "2026-04-27")  # Standarddatum für Testdaten
    message = None

    connection = get_db_connection()

    patient = connection.execute(""" 
        SELECT * FROM patients WHERE id = ? 
    """, (patient_id,)).fetchone()

    if patient is None:
        connection.close()
        return "Patient wurde nicht gefunden."

    if request.method == "POST":
        record_date = request.form["record_date"]
        record_time = request.form["record_time"]
        blood_pressure = request.form["blood_pressure"]
        pulse = request.form["pulse"]
        temperature = request.form["temperature"]
        oxygen_saturation = request.form["oxygen_saturation"]
        note = request.form["note"]

        connection.execute(""" 
            INSERT INTO vital_signs ( 
                patient_id, 
                recorded_by, 
                record_date, 
                record_time, 
                blood_pressure, 
                pulse, 
                temperature, 
                oxygen_saturation, 
                note 
            ) 
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?) 
        """, (
            patient_id,
            session["username"],
            record_date,
            record_time,
            blood_pressure,
            pulse,
            temperature,
            oxygen_saturation,
            note
        ))

        connection.commit()
        selected_date = record_date
        message = "Vitalwerte wurden gespeichert."

    vital_list = connection.execute(""" 
        SELECT * FROM vital_signs 
        WHERE patient_id = ? AND record_date = ? 
        ORDER BY record_time 
    """, (patient_id, selected_date)).fetchall()

    connection.close()

    curve_rows = []

    pulse_values = []
    temperature_values = []
    spo2_values = []

    for vital in vital_list:
        status = "normal"
        alerts = []

        pulse = vital["pulse"]
        temperature = vital["temperature"]
        oxygen_saturation = vital["oxygen_saturation"]

        if pulse is not None and pulse != "":
            pulse = int(pulse)
            pulse_values.append(pulse)

            if pulse > 100:
                status = "auffällig"
                alerts.append("Puls erhöht")

        if temperature is not None and temperature != "":
            temperature = float(temperature)
            temperature_values.append(temperature)

            if temperature >= 38.0:
                status = "auffällig"
                alerts.append("Temperatur erhöht")

        if oxygen_saturation is not None and oxygen_saturation != "":
            oxygen_saturation = int(oxygen_saturation)
            spo2_values.append(oxygen_saturation)

            if oxygen_saturation < 94:
                status = "auffällig"
                alerts.append("SpO₂ niedrig")

        curve_rows.append({
            "data": vital,
            "status": status,
            "alerts": ", ".join(alerts) if alerts else "Keine Auffälligkeit"
        })

    abnormal_count = 0

    for row in curve_rows:
        if row["status"] == "auffällig":
            abnormal_count += 1

    summary = {
        "entries": len(curve_rows),
        "abnormal_count": abnormal_count,
        "max_pulse": max(pulse_values) if pulse_values else "-",
        "max_temperature": max(temperature_values) if temperature_values else "-",
        "min_spo2": min(spo2_values) if spo2_values else "-"
    }

    return render_template(
        "patient_curve.html",
        patient=patient,
        curve_rows=curve_rows,

@app.route("/nursing_tasks")  # Aufgabenübersicht für Pflegekräfte

    if "username" not in session:
        return redirect(url_for("login"))

    if session["role"] not in ["nurse", "admin"]:
        return "Zugriff verweigert. Nur Pflegekräfte dürfen die Aufgabenübersicht sehen."

    connection = get_db_connection()

    open_orders = connection.execute(""" 
        SELECT doctor_orders.*, patients.name AS patient_name 
        FROM doctor_orders 
        JOIN patients ON doctor_orders.patient_id = patients.id 
        WHERE doctor_orders.status = 'offen' 
        ORDER BY doctor_orders.created_at DESC 
    """).fetchall()

    open_medications = connection.execute(""" 
        SELECT medication_plan.*, patients.name AS patient_name 
        FROM medication_plan 
        JOIN patients ON medication_plan.patient_id = patients.id 
        WHERE medication_plan.status = 'offen' 
        ORDER BY medication_plan.schedule_time 
    """).fetchall()

    connection.close()

    return render_template(
        "nursing_tasks.html",
        open_orders=open_orders,
        open_medications=open_medications
    )
